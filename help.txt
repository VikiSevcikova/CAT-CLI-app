node app.js --help                                       Show supported params
node app.js --version                                    See version

node app.js --createFile "file.txt"                     Create an empty file called file.txt
node app.js --writeFile "file.txt" "text"               Create file file.txt if not exists and write data, if exists and there is a data then it is overwritten
node app.js --appendFile "file.txt" "text"              Write data to the file.txt, if there is data, then append to the end
node app.js --renameFile "from.txt" "to.txt"            Rename a file
node app.js --deleteFile "file.txt"                     Delete file
node app.js --listFiles "directory"                     Show list of files in the "directory"
node app.js --copyFile "source.txt" "copy.txt"          Copy file, it the name of the copy file is missing, postfix _copy is added to the source file's name
node app.js --movefile "from/file.txt" "to/file.txt"    Move file to the different directory
node app.js --size "file.txt"                           Show the size of the file
node app.js --view "somefile"                           Show the content of the file
node app.js --view "somefile" --pause "22"              Show the content of the file, after 22 lines make a pause
